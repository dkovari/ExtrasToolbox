105  // does not change memory
106  virtual void updateDims(const std::vector<size_t>& dim){
107  bool sameDim = true;
108  if(_dims.size()!=dim.size()){
109  sameDim = false;
110  }
111  for(size_t n=0;(n<dim.size()&&sameDim);++n){
112  if(_dims[n]!=dim[n]){
113  sameDim = false;
114  }
115  }
116  if(!sameDim){
117  computePagesz(dim);
118 
119  if(dim.empty()){
120  _numel = 0;
121  _nRows = 0;
122  _nCols = 0;
123  }
124  else{
125  _numel = _pagesz.back()*dim.back();
126  _nRows = dim[0];
127  _nCols = _numel/_nRows;
128  }
129  _dims = dim;
130  }
131  }

133  virtual void computePagesz(const std::vector<size_t>& dim){
134  _pagesz.resize(dim.size());
135  if(!dim.empty()){
136  _pagesz[0] = 1;
137  for(size_t n=1;n<dim.size();++n){
138  _pagesz[n] = _pagesz[n-1]*dim[n-1];
139  }
140  }
141  }

196  virtual void reallocdata(const std::vector<size_t>& dim){
197 
198  if(_numel==cumprod(dim)){ //size not changing, just update dims
199  updateDims(dim);
200  }
201  else if(!_managedata){ //not managing data, copy
202  size_t new_numel = cumprod(dim);
203  T* newdata = (T*)malloc(new_numel*sizeof(T));
204  if(new_numel<_numel){
205  memcpy(newdata,_data,new_numel*sizeof(T));
206  }else{
207  memcpy(newdata,_data,_numel*sizeof(T));
208  }
209  _managedata = true;
210  updateDims(dim);
211  }
212  else { //managing data, realloc
213  updateDims(dim);
214  if (_data != nullptr) {
215  _data = (T*)realloc(_data, _numel * sizeof(T));
216  }
217  else
218  {
219  _data = (T*)malloc(_numel * sizeof(T));
220  }
221  }
222  }

627  template <typename M>
628  NumericArray& concatenate(const ArrayBase<M>& b, size_t dim) {
629  if (iscomplex()) { //error on complex data ---------> change in the future
630  throw(extras::stacktrace_error("NumericArray::concatenate(): cat for complex data not implemented."));
631  }
632  if (isConst()) { //self-manage the data
633  selfManage();
634  }
635 
637  // Compute new size
638  auto thisSz = size();
639  auto thatSz = b.size();
640  size_t thisSz_len = thisSz.size();
641  size_t thatSz_len = thatSz.size();
642 
643  size_t maxDimLen = std::max(std::max(thisSz.size(), thatSz.size()), dim + 1);
644 
645  // Loop over array dimensions and determine if sizes are compatible
646  thisSz.resize(maxDimLen);
647  thatSz.resize(maxDimLen);
648 
649  for (size_t j = 0; j < maxDimLen; j++) {
650 
651  if (j >= thisSz_len) {
652  thisSz[j] = 1;
653  }
654 
655  if (j >= thatSz_len) {
656  thatSz[j] = 1;
657  }
658 
659  if (j == dim) {
660  continue;
661  }
662 
663  if (thisSz[j] != thatSz[j]) {
664  throw(extras::stacktrace_error("incompatible sizes for concatenate"));
665  }
666  }
667 
668  auto newSz = thisSz;
669  newSz[dim] = thisSz[dim] + thatSz[dim];
670 
672  // Create Temporary array to hold new data
673 
674  NumericArray<T> newArray(newSz);
675 
677  // Loop over dimension dim and copy data for all other dimensions
678 
679  size_t new_dim = 0; //accumulator for current position in newPtr
680  for (size_t d = 0; d < thisSz[dim]; d++) { //first loop over mxptr elements
681  //loop over other dims
682  for (size_t odim = 0; odim < newSz.size(); odim++) {
683  if (odim == dim) { //nothing to do for dim
684  continue;
685  }
686  if (thisSz[odim] == 1) {//nothing to do for singleton dimension
687  continue;
688  }
689  //loop over indecies of odim and copy
690  for (size_t od = 0; od < thisSz[odim]; od++) {
691  std::vector<size_t> subs(newSz.size(), 0.0); //create subscript array with all zeros
692  subs[odim] = od; //index for dimension being copied
693  subs[dim] = d; //index for dimension being concatenated
694 
695  newArray(subs) = (*this)(subs); //copy values
696 
697  }
698  }
699  new_dim++;
700  }
701 
702  for (size_t d = 0; d < thatSz[dim]; d++) { //second loop over end_obj elements
703  //loop over other dims
704  for (size_t odim = 0; odim < newSz.size(); odim++) {
705  if (odim == dim) { //nothing to do for dim
706  continue;
707  }
708  if (thatSz[odim] == 1) {//nothing to do for singleton dimension
709  continue;
710  }
711  //loop over indecies of odim and copy
712  for (size_t od = 0; od < thatSz[odim]; od++) {
713  std::vector<size_t> subs(newSz.size(), 0.0); //create subscript array with all zeros
714  subs[odim] = od; //index for dimension being copied
715  subs[dim] = d; //index for dimension being concatenated
716 
717  auto val = b(subs); //make tmp copy of data
718 
719  subs[dim] = new_dim; //change the sub, since newArray has different dim
720  newArray(subs) = val; //set value
721  }
722  }
723  new_dim++;
724  }
725 
726  // Set cleanup
727 
728  mxFree(mxGetData(getmxarray()));
729  mxSetData(getmxarray(),mxGetData(newArray.getmxarray()));
730  mxSetData(newArray.getmxarray(), nullptr);
731  mxSetDimensions(getmxarray(), newSz.data(), newSz.size()); //change dimensions
732 
733  if (isPersistent()) {
734  mexMakeMemoryPersistent(getdata());
735  }
736 
737  return *this;
738  }

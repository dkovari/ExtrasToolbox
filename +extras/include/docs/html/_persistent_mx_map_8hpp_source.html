101  virtual void setParameters(size_t nrhs, const mxArray* prhs[]) {
102 
103  //mexPrintf("Inside ParameterMxMap::setParameters()\n");
104  //mexEvalString("pause(0.2)");
105 
106  if (_setfromstruct && nrhs==1) { // set from struct allowed
107  if (!mxIsStruct(prhs[0])) {
108  throw(std::runtime_error("ParameterMxMap::setParameters(): a single mxArray* was passed, but is was not a struct."));
109  }
110 
111  if (!mxGetNumberOfElements(prhs[0]) != 1) {
112  throw(std::runtime_error("ParameterMxMap::setParameters(): struct array must be a scalar struct (i.e. numel==1)"));
113  }
114 
115  MxStruct thisStruct(prhs[0]);
116  auto fnames = thisStruct.fieldnames();
117  for (size_t n = 0; n < thisStruct.number_of_fields(); n++) {
118  (*this)[fnames[n]] = thisStruct(0, fnames[n].c_str());
119  }
120  }
121 
122  if (nrhs % 2 != 0) {
123  throw(std::runtime_error("ParameterMxMap::setParameters() number of args must be even (specified as Name,Value pairs)."));
124  }
125 
126  // loop over args and set parameters
127  for (size_t n = 0; n < nrhs - 1; n += 2) {
128  (*this)[extras::cmex::getstring(prhs[n])] = extras::cmex::persistentMxArray(prhs[n + 1]);
129  }
130 
131  }

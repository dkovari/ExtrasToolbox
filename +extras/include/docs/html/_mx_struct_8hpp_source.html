191  FieldWrapper(MxStruct& parent, size_t idx, const char* fieldname) :
192  _parent(parent),
193  index(idx)
194  {
195  field_number = mxGetFieldNumber(_parent.getmxarray(), fieldname);
196  if (field_number<0) { //need to create field
197 
198  if (_parent.isConst()) {
199  throw(extras::stacktrace_error("FieldWrapper: Cannot add field to const mxArray*"));
200  }
201 
202  field_number = mxAddField(_parent.getmxarray(), fieldname);
203  }
204 
205  if (field_number<0) {
206  throw(extras::stacktrace_error(std::string("FieldWrapper(): Could not create fieldname:") + std::string(fieldname)));
207  }
208  }

225  if (_parent.isConst()) {

228  mxDestroyArray(mxGetFieldByNumber(_parent.getmxarray(), index, field_number));
229  mxSetFieldByNumber(_parent.getmxarray(), index, field_number, src);
230  }

233  if (_parent.isConst()) {
234  throw(extras::stacktrace_error("FieldWrapper: Cannot return mxArray* for specified field because parent cell is const."));
235  }
236  return mxGetFieldByNumber(_parent.getmxarray(), index, field_number);
237  }

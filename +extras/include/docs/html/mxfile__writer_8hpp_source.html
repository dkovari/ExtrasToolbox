60  std::list<SerialData> Serialize(size_t nrhs, const mxArray** prhs) {
61  std::list<SerialData> out;
62  for (size_t i = 0; i < nrhs; i++) {
63  SerialData l;
64 
65  if (prhs[i] == nullptr) { //passed nullptr, interpret as empty numeric array
66  l.nbytes = sizeof(uint8_t) + sizeof(size_t) + sizeof(uint8_t) + sizeof(uint8_t); //type, ndims, isComplex, isInterleaved,
67  l.nbytes += 0; //mxGetNumberOfDimensions(prhs[i]) * sizeof(size_t) + mxGetNumberOfElements(prhs[i]) * mxGetElementSize(prhs[i]) * (1 + mxIsComplex(prhs[i])); // dims, data
68  l.data = prhs[i];
69  out.push_back(l);
70  continue;
71  }
72 
73  switch (mxGetClassID(prhs[i])) {
74  case mxCELL_CLASS:
75  l.nbytes = sizeof(uint8_t) + sizeof(size_t) + sizeof(size_t) * mxGetNumberOfDimensions(prhs[i]);
76  //type, ndim, dims
77  l.data = prhs[i];
78  out.push_back(l);
79  for (size_t j = 0; j < mxGetNumberOfElements(prhs[i]); ++j) {
80 
81  const mxArray * c = mxGetCell(prhs[i], j);
82 
83  out.splice(out.end(), Serialize(1, &c));
84 
85  }
86  break;
87  case mxSTRUCT_CLASS: {
88  l.nbytes = sizeof(uint8_t) + sizeof(size_t) + mxGetNumberOfDimensions(prhs[i]) + sizeof(std::string);
89  for (size_t j = 0; j < mxGetNumberOfFields(prhs[i]); ++j) {
90  l.nbytes += sizeof(char) * std::string(mxGetFieldNameByNumber(prhs[i], j)).length();
91  }
92  //type, ndims, dims, nfields, field names
93  l.data = prhs[i];
94  out.push_back(l);
95  for (size_t j = 0; j < mxGetNumberOfElements(prhs[i]); ++j) {
96  for (size_t k = 0; k < mxGetNumberOfFields(prhs[i]); ++k) {
97  const mxArray* c = mxGetFieldByNumber(prhs[i], j, k);
98  out.splice(out.end(), Serialize(1, &c)); //unfold contents of fields
99  }
100  }
101  }
102  break;
103  default:
104  l.nbytes = sizeof(uint8_t) + sizeof(size_t) + sizeof(uint8_t) + sizeof(uint8_t); //type, ndims, isComplex, isInterleaved,
105  l.nbytes += mxGetNumberOfDimensions(prhs[i]) * sizeof(size_t) + mxGetNumberOfElements(prhs[i]) * mxGetElementSize(prhs[i]) * (1 + mxIsComplex(prhs[i])); // dims, data
106  l.data = prhs[i];
107  out.push_back(l);
108  }
109  }
110  return out;
111  }

241  void writeList(const std::list<SerialData>& dataList, SimpleWriter& FP) {
242  for (auto& thisData : dataList) { //loop over all items in the list
243  const mxArray* thisArray = thisData.data;
244  uint8_t type; //= mxGetClassID(thisArray);
245  size_t ndims; //= mxGetNumberOfDimensions(thisArray);
246  void* dims = nullptr;
247 
248  if (thisArray == nullptr) { //passed nullptr, interpret as empty numeric array
249  type = mxDOUBLE_CLASS;
250  ndims = 0;
251  }
252  else {
253  type = mxGetClassID(thisArray);
254  ndims = mxGetNumberOfDimensions(thisArray);
255  dims = (void*)mxGetDimensions(thisArray);
256  }
257 
259 
260  FP.write(&type, 1);//write type byte
261  FP.write(&ndims, sizeof(size_t));//write ndim
262  FP.write(dims, sizeof(size_t)*ndims);//write dims
263 
264  // Write datatype specific info
265  switch (type)
266  {
267  case mxCELL_CLASS:
268  //nothing else to write
269  break;
270  case mxSTRUCT_CLASS:
271  {
272  //write number of field names
273  size_t nfields = mxGetNumberOfFields(thisArray);
274  FP.write(&nfields, sizeof(size_t));
275 
277  // FOR EACH FIELD
278  //write length of fieldname and fieldnames
279  for (size_t f = 0; f < nfields; ++f) {
280  const char* fieldname = mxGetFieldNameByNumber(thisArray, f);
281  size_t len = strlen(fieldname) + 1; // length including null terminator
282 
283  //write length of name
284  FP.write(&len, sizeof(size_t));
285 
286  //write fieldname, including null terminator
287  FP.write(fieldname, len * sizeof(char));
288  }
289  }
290  break;
291  default: //all other types
292  {
293  if (thisArray == nullptr) { //passed nullptr, interpret as empty numeric array
294  //write complex
295  uint8_t isComplex = 0;
296  FP.write(&isComplex, sizeof(uint8_t));
297 
298  //write interleaved flag
299  uint8_t interFlag = 0; //default to not interleaved complex data
300 #if MX_HAS_INTERLEAVED_COMPLEX
301  interFlag = 1;
302 #endif
303  FP.write(&interFlag, sizeof(uint8_t));
304 
305  //Write data
306  //nothing to do since there is no data
307  }
308  else
309  {
310  //write complex
311  uint8_t isComplex = (uint8_t)mxIsComplex(thisArray);
312  FP.write(&isComplex, sizeof(uint8_t));
313 
314  //write interleaved flag
315  uint8_t interFlag = 0; //default to not interleaved complex data
316 #if MX_HAS_INTERLEAVED_COMPLEX
317  interFlag = 1;
318 #endif
319  FP.write(&interFlag, sizeof(uint8_t));
320 
321  //Write data
322  size_t numel = mxGetNumberOfElements(thisArray);
323  size_t elsz = mxGetElementSize(thisArray);
324  if (!isComplex) { //not complex, simple write
325  FP.write(mxGetData(thisArray), numel*elsz);
326  }
327  else { //is complex
328 #if MX_HAS_INTERLEAVED_COMPLEX //interleaved data, size is 2x so regular copy works fine
329  FP.write(mxGetData(thisArray), numel*elsz);
330 #else //not interleaved, need to write imag data explicitly
331  FP.write(mxGetData(thisArray), numel*elsz);
332  FP.write(mxGetImagData(thisArray), numel*elsz);
333 #endif
334  }
335  }
336  }
337  }
338 
339  // force file write
340  //FP.flush();
341  }

347  class MxFileWriter {
348  protected:

352 
354  std::string validateFileExt(std::string fpth) {
355  //look for ".mxf.gz" at end of file
356  std::string fpth_lower = extras::tolower(fpth);
357  const char* fpth_c = fpth_lower.c_str();
358  const char* p_ext = strstr(fpth_c, ".mxf.gz");
359  if (p_ext != nullptr) { //found in fpth, make sure it's at the end
360  size_t loc = p_ext - fpth_c; //location in the string
361  size_t back = fpth.size() - loc; //loc from back
362  if (back == 7) { //found at back, just return fpth
363  return fpth;
364  }
365  }
366  //didn't find ".mxf.gz", look for ".mxf"
367  p_ext = strstr(fpth_c, ".mxf");
368  if (p_ext != nullptr) { //found in fpth, make sure it's at the end
369  size_t loc = p_ext - fpth_c; //location in the string
370  size_t back = fpth.size() - loc; //loc from back
371  if (back == 7) { //found at back, add ".gz"
372  fpth += ".gz";
373  return fpth;
374  }
375  }
376  //didn't find ".mxf" add ".mxf.gz"
377  fpth += ".mxf.gz";
378  return fpth;
379  }
380  public:
381 
383  bool isFileOpen() const {
384  return _WritePointer.isopen();
385  }
386

406  virtual void openFile(std::string fpth) {
407  closeFile();
408  std::lock_guard<std::mutex> lock(_WPmutex);
409  auto fp_ext = validateFileExt(fpth);
410  _WritePointer.open(fp_ext.c_str());
411  _filepath = fp_ext;
412  }

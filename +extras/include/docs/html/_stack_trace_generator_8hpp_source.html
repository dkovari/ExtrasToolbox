32 class StackTraceGenerator
33 {
34 private:
35 
36  // this is a pure utils class
37  // cannot be instantiated
38  //
39  StackTraceGenerator() = delete;
40  StackTraceGenerator(const StackTraceGenerator&) = delete;
41  StackTraceGenerator& operator=(const StackTraceGenerator&) = delete;
42  ~StackTraceGenerator() = delete;
43 
44 public:
45 
46  static std::vector<std::string> GetTrace()
47  {
48  // record stack trace upto 128 frames
49  int callstack[128] = {};
50 
51  // collect stack frames
52  int frames = backtrace((void**)callstack, 128);
53 
54  // get the human-readable symbols (mangled)
55  char** strs = backtrace_symbols((void**)callstack, frames);
56 
57  std::vector<std::string> stackFrames;
58  stackFrames.reserve(frames);
59 
60  for(int i = 0; i < frames; ++i)
61  {
62  char functionSymbol[1024] = {};
63  char moduleName[1024] = {};
64  int offset = 0;
65  char addr[48] = {};
66 
67  /*
68 
69  Typically this is how the backtrace looks like:
70 
71  0 <app/lib-name> 0x0000000100000e98 _Z5tracev + 72
72  1 <app/lib-name> 0x00000001000015c1 _ZNK7functorclEv + 17
73  2 <app/lib-name> 0x0000000100000f71 _Z3fn0v + 17
74  3 <app/lib-name> 0x0000000100000f89 _Z3fn1v + 9
75  4 <app/lib-name> 0x0000000100000f99 _Z3fn2v + 9
76  5 <app/lib-name> 0x0000000100000fa9 _Z3fn3v + 9
77  6 <app/lib-name> 0x0000000100000fb9 _Z3fn4v + 9
78  7 <app/lib-name> 0x0000000100000fc9 _Z3fn5v + 9
79  8 <app/lib-name> 0x0000000100000fd9 _Z3fn6v + 9
80  9 <app/lib-name> 0x0000000100001018 main + 56
81  10 libdyld.dylib 0x00007fff91b647e1 start + 0
82 
83  */
84 
85  // split the string, take out chunks out of stack trace
86  // we are primarily interested in module, function and address
87  sscanf(strs[i], "%*s %s %s %s %*s %d",
88  &moduleName, &addr, &functionSymbol, &offset);
89 
90  int validCppName = 0;
91  // if this is a C++ library, symbol will be demangled
92  // on success function returns 0
93  //
94  char* functionName = abi::__cxa_demangle(functionSymbol,
95  NULL, 0, &validCppName);
96 
97  char stackFrame[4096] = {};
98  if (validCppName == 0) // success
99  {
100  sprintf(stackFrame, "(%s)\t0x%s — %s + %d",
101  moduleName, addr, functionName, offset);
102  }
103  else
104  {
105  // in the above traceback (in comments) last entry is not
106  // from C++ binary, last frame, libdyld.dylib, is printed
107  // from here
108  sprintf(stackFrame, "(%s)\t0x%s — %s + %d",
109  moduleName, addr, functionName, offset);
110  }
111 
112  if (functionName)
113  {
114  free(functionName);
115  }
116 
117  std::string frameStr(stackFrame);
118  stackFrames.push_back(frameStr);
119  }
120  free(strs);
121 
122  return stackFrames;
123  }
124 };

118  extras::cmex::MxObject readNext(FILE_ReadPointer& FP,bool calledRecursively=false) {
119  extras::cmex::MxObject out;
120 
121  // read type
122  mxClassID thisClassID;
123  uint8_t tmp;
124  if (FP.read(&tmp, 1) < 1) {
125  if (FP.eof()) {
126  if (calledRecursively) {
127  throw(extras::stacktrace_error("readNext(): at EOF and called recursively"));
128  }
129  return (mxArray*)nullptr;//return empty array;
130  }
131  else {
132  throw(extras::stacktrace_error(FP.error_msg()));
133  }
134  }
135  thisClassID = (mxClassID)tmp;
136 
137  //read ndim
138  size_t ndim;
139  if (FP.read(&ndim, sizeof(size_t)) < sizeof(size_t)) {
140  if (FP.eof()) {
141  throw(extras::stacktrace_error("reached end of file while reading ndim"));
142  }
143  else {
144  throw(extras::stacktrace_error(FP.error_msg()));
145  }
146  }
147 
148  //read dims
149  std::vector<size_t> dims(ndim);
150  if (FP.read(dims.data(), sizeof(size_t)*ndim) < sizeof(size_t)*ndim) {
151  if (FP.eof()) {
152  throw(extras::stacktrace_error("reached end of file while reading dims"));
153  }
154  else {
155  throw(extras::stacktrace_error(FP.error_msg()));
156  }
157  }
158 
159  //create array to hold data
160  switch (thisClassID) {
161  case mxCELL_CLASS:
162  {
163  out.own(mxCreateCellArray(ndim, dims.data()));
164  for (size_t n = 0; n < out.numel(); n++) {
165  mxSetCell(out.getmxarray(), n, readNext(FP,true));
166  }
167  }
168  break;
169  case mxSTRUCT_CLASS:
170  {
171  //get nfields
172  size_t nfields;
173  if (FP.read(&nfields, sizeof(size_t)) < sizeof(size_t)) {
174  if (FP.eof()) {
175  throw(extras::stacktrace_error("reached end of file while reading nfields"));
176  }
177  else {
178  throw(extras::stacktrace_error(FP.error_msg()));
179  }
180  }
181 
182  std::vector<std::vector<char>> fieldnames(nfields); //vector of strings in which names are stored
183  std::vector<const char*> fnames(nfields); //vector of const char* used to call mxCreateStruct...
184  for (size_t f = 0; f < nfields; f++) { //loop and read names
185  size_t len; //length of fieldname including null terminator
186  if (FP.read(&len, sizeof(size_t)) < sizeof(size_t)) {
187  if (FP.eof()) {
188  throw(extras::stacktrace_error("reached end of file while reading length of fieldname"));
189  }
190  else {
191  throw(extras::stacktrace_error(FP.error_msg()));
192  }
193  }
194 
195  fieldnames[f].resize(len); //read fieldname, including null terminator
196  if (FP.read(fieldnames[f].data(), len) < len) {
197  if (FP.eof()) {
198  throw(extras::stacktrace_error("reached end of file while reading lfieldname"));
199  }
200  else {
201  throw(extras::stacktrace_error(FP.error_msg()));
202  }
203  }
204  fnames[f] = fieldnames[f].data(); //set pointer array
205  }
206 
207  out.own(mxCreateStructArray(ndim, dims.data(), nfields, fnames.data()));
208 
210  for (size_t j = 0; j < out.numel(); ++j) {
211  for (size_t k = 0; k < nfields; ++k) {
212  mxSetFieldByNumber(out.getmxarray(), j, k, readNext(FP,true));
213  }
214  }
215 
216  }
217  break;
218  default:
219  {
220  //read complexity
221  uint8_t isComplex;
222  if (FP.read(&isComplex, 1) < 1) {
223  if (FP.eof()) {
224  throw(extras::stacktrace_error("reached end of file while reading isComplex"));
225  }
226  else {
227  throw(extras::stacktrace_error(FP.error_msg()));
228  }
229  }
230 
231  //read interleaved
232  uint8_t interFlag;
233  if (FP.read(&interFlag, 1) < 1) {
234  if (FP.eof()) {
235  throw(extras::stacktrace_error("reached end of file while reading interFlag"));
236  }
237  else {
238  throw(extras::stacktrace_error(FP.error_msg()));
239  }
240  }
241 
242  if (thisClassID == mxCHAR_CLASS) { //handle char
243  out.own(mxCreateCharArray(ndim, dims.data()));
244  size_t elsz = out.elementBytes();
245  if (FP.read(mxGetData(out.getmxarray()), elsz*out.numel()) < elsz*out.numel()) {
246  if (FP.eof()) {
247  throw(extras::stacktrace_error("reached end of file while reading char data"));
248  }
249  else {
250  throw(extras::stacktrace_error(FP.error_msg()));
251  }
252  }
253  }
254  else if (!isComplex) { //real
255  out.own(mxCreateNumericArray(ndim, dims.data(), thisClassID, mxREAL));
256  size_t elsz = out.elementBytes();
257  if (FP.read(mxGetData(out.getmxarray()), elsz*out.numel()) < elsz*out.numel()) {
258  if (FP.eof()) {
259  throw(extras::stacktrace_error("reached end of file while reading real numeric data"));
260  }
261  else {
262  throw(extras::stacktrace_error(FP.error_msg()));
263  }
264  }
265  }
266  else { //complex
267  size_t elsz_real = cmex::elementBytes(thisClassID, false);
268  size_t numel = prod(dims);
269  if (interFlag) { //data is interleaved
270  char* data = (char*)mxMalloc(numel*elsz_real * 2);
271  if (FP.read(data, numel*elsz_real * 2) < numel*elsz_real * 2) {
272  if (FP.eof()) {
273  mxFree(data); //cleanup out memory mess
274  throw(extras::stacktrace_error("reached end of file while reading interleaved complex numeric data"));
275  }
276  else {
277  mxFree(data); //cleanup out memory mess
278  throw(extras::stacktrace_error(FP.error_msg()));
279  }
280  }
281  // create array and set data
282  mxArray* tmp = mxCreateNumericMatrix(0, 0, thisClassID, mxCOMPLEX);
283  mxSetDimensions(tmp, dims.data(), ndim);
284 #if MX_HAS_INTERLEAVED_COMPLEX //interleaved, just do a regular set
285  mxSetData(tmp, data);
286 #else //data was interleaved, but MATLAB is not
287  char* realData = (char*)mxMalloc(numel*elsz_real);
288  char* imagData = (char*)mxMalloc(numel*elsz_real);
289 
290  //copy elements
291  for (size_t n = 0; n < numel; n++) {
292  std::memcpy(&(realData[n*elsz_real]), &(data[(n + 0)*elsz_real]), elsz_real); //copy real element
293  std::memcpy(&(imagData[n*elsz_real]), &(data[(n + 1)*elsz_real]), elsz_real); //copy imag element
294  }
295  mxSetData(tmp, realData); //set real data
296  mxSetImagData(tmp, imagData); //set imag data
297  mxFree(data); //free data holding interleaved, we arent using it anymore
298 #endif
299  out.own(tmp);
300  }
301  else { //data is not interleaved
302  char* realData = (char*)mxMalloc(numel*elsz_real);
303  char* imagData = (char*)mxMalloc(numel*elsz_real);
304  if (FP.read(realData, numel*elsz_real) < numel*elsz_real) { //read real
305  if (FP.eof()) {
306  mxFree(realData); //cleanup out memory mess
307  mxFree(imagData); //cleanup out memory mess
308  throw(extras::stacktrace_error("reached end of file while reading non-interleaved real data for complex numeric"));
309  }
310  else {
311  mxFree(realData); //cleanup out memory mess
312  mxFree(imagData); //cleanup out memory mess
313  throw(extras::stacktrace_error(FP.error_msg()));
314  }
315  }
316  if (FP.read(imagData, numel*elsz_real) < numel*elsz_real) { //read imag
317  if (FP.eof()) {
318  mxFree(realData); //cleanup out memory mess
319  mxFree(imagData); //cleanup out memory mess
320  throw(extras::stacktrace_error("reached end of file while reading non-interleaved imag data for complex numeric"));
321  }
322  else {
323  mxFree(realData); //cleanup out memory mess
324  mxFree(imagData); //cleanup out memory mess
325  throw(extras::stacktrace_error(FP.error_msg()));
326  }
327  }
328 
329  mxArray* tmp = mxCreateNumericMatrix(0, 0, thisClassID, mxCOMPLEX);
330  mxSetDimensions(tmp, dims.data(), ndim);
331 
332 #if MX_HAS_INTERLEAVED_COMPLEX //data was not interleaved, but MATLAB is
333  char* interData = (char*)mxMalloc(numel*elsz_real * 2);
334  //copy elements
335  for (size_t n = 0; n < numel; n++) {
336  std::memcpy(&(interData[(n + 0)*elsz_real]),&(realData[n*elsz_real]),elsz_real); //copy real element
337  std::memcpy(&(interData[(n + 1)*elsz_real]),&(imagData[n*elsz_real]), elsz_real); //copy imag element
338  }
339 
340  mxSetData(tmp, interData);
341  mxFree(realData);
342  mxFree(imagData);
343 #else //both are not interleaved
344  mxSetData(tmp, realData);
345  mxSetImagData(tmp, imagData);
346 #endif
347  out.own(tmp);
348 
349  }
350  }
351  }
352  }
353 
354  return out;
355  }

403  void openFile(std::string fpth) {
404  closeFile();
405 
406  //determine size of file
407  FILE* fp = fopen(fpth.c_str(), "rb");
408  if (fp == nullptr) {
409  throw(extras::stacktrace_error(std::string("MxFileReader::openFile(): returned null, file:'") + std::string(fpth) + std::string("' could not be opened.")));
410  }
411  if (fseek(fp, 0, SEEK_END) != 0) {
412  fclose(fp);
413  throw(extras::stacktrace_error(std::string("MxFileReader::openFile(): file:'") + std::string(fpth) + std::string("' Could not determine file size.")));
414  }
415  _compressedSize = ftell(fp);
416  fclose(fp); //close
417 
418  gzFile gzf = gzopen(fpth.c_str(), "rb");
419  if (gzf == NULL) {
420  throw(extras::stacktrace_error(std::string("MxFileReader::openFile(): returned null, file:'") + std::string(fpth) + std::string("' could not be opened.")));
421  }
422  gzbuffer(gzf, 262144);
423  std::lock_guard<std::mutex> lock(_RPmutex);
424  _ReadPointer = GZFILE_ReadPointer(gzf);
425  _filepath = fpth;
426  _currentCompressedPosition = _ReadPointer.getOffset();
427  }

438  extras::cmex::MxObject readNextArray() {
439  if (!isFileOpen()) {
440  throw(extras::stacktrace_error(std::string("MxFileReader::readNextArray() file:'") + _filepath + std::string("' is not open.")));

479  template<class ObjType, extras::SessionManager::ObjectManager<ObjType>& ObjManager> /*ObjType should be a derivative of MxFileReader*/
480  class MxFileReaderInterface : public SessionManager::mexInterface<ObjType, ObjManager> {
481  typedef SessionManager::mexInterface<ObjType, ObjManager> ParentType;
482  protected:
483  void openFile(int nlhs, mxArray* plhs[], int nrhs, const mxArray* prhs[]) {
484  ParentType::getObjectPtr(nrhs, prhs)->openFile(nrhs - 1, &prhs[1]);
485  }
486  void closeFile(int nlhs, mxArray* plhs[], int nrhs, const mxArray* prhs[]) {
487  ParentType::getObjectPtr(nrhs, prhs)->closeFile();
488  }
489  void filepath(int nlhs, mxArray* plhs[], int nrhs, const mxArray* prhs[]) {
490  cmex::MxObject fpth = ParentType::getObjectPtr(nrhs, prhs)->filepath();
491  plhs[0] = fpth;
492  }
493  void isEOF(int nlhs, mxArray* plhs[], int nrhs, const mxArray* prhs[]) {
494  bool iseof = ParentType::getObjectPtr(nrhs, prhs)->isEOF();
495  plhs[0] = mxCreateLogicalScalar(iseof);
496  }
497  void readNextArray(int nlhs, mxArray* plhs[], int nrhs, const mxArray* prhs[]) {
498  cmex::MxObject out = ParentType::getObjectPtr(nrhs, prhs)->readNextArray();
499  if (out.getmxarray() == nullptr) {
500  if (ParentType::getObjectPtr(nrhs, prhs)->isEOF()) {
501  plhs[0] = mxCreateCellMatrix(0, 0);
502  return;
503  }
504  throw(extras::stacktrace_error(std::string("MxFileReader::readNextArray() File:'") + ParentType::getObjectPtr(nrhs, prhs)->filepath() + std::string("'. Recieved Nullptr and not at EOF.")));
505  }
506  plhs[0] = out;
507  }
508  void isFileOpen(int nlhs, mxArray* plhs[], int nrhs, const mxArray* prhs[]) {
509  bool isopen = ParentType::getObjectPtr(nrhs, prhs)->isFileOpen();
510  plhs[0] = mxCreateLogicalScalar(isopen);
511  }
512  void getCompressedSize(int nlhs, mxArray* plhs[], int nrhs, const mxArray* prhs[]) {
513  size_t val = ParentType::getObjectPtr(nrhs, prhs)->getCompressedSize();
514  plhs[0] = mxCreateDoubleScalar(val);
515  }
516  void getPositionInFile(int nlhs, mxArray* plhs[], int nrhs, const mxArray* prhs[]) {
517  size_t val = ParentType::getObjectPtr(nrhs, prhs)->getPositionInFile();
518  plhs[0] = mxCreateDoubleScalar(val);
519  }
520  void loadProgress(int nlhs, mxArray* plhs[], int nrhs, const mxArray* prhs[]) {
521  size_t val = ParentType::getObjectPtr(nrhs, prhs)->loadProgress();
522  plhs[0] = mxCreateDoubleScalar(val);
523  }

545  class AsyncMxFileReader {
546  protected:
547  mutable std::mutex _ResultsGroupMutex; //mutex lock for _ResultsGroup
548  cmex::mxArrayGroup _ResultsGroup; //protected by _ResultsGroupMutex
549 #ifdef _DEBUG
550  std::vector<const mxArray*> _ResPtrs;
551  std::vector<const mxArray*> _ResultsPtrs;
552 #endif
553 
554  MxFileReader _Reader; //read for loading data from file
555 
556  std::thread mThread;
557  std::atomic_bool ProcessRunning = false; //flag for stopping to processing loop
558  std::atomic_bool _atEOF = false;
559 
560  std::atomic_bool ErrorFlag;
561  std::mutex LastErrorMutex;
562  std::exception_ptr LastError = nullptr;
563 
564  void LoadDataLoop() {
565  try {
566  while (_Reader.isFileOpen() && ProcessRunning) {
567  if (_Reader.isEOF()) {
568  _atEOF = true;
569  ProcessRunning = false;
570  return;
571  }
572  cmex::MxObject res = _Reader.readNextArray(); //read next data
573  if (res.getmxarray() == nullptr) {
574  if (_Reader.isEOF()) {
575  _atEOF = true;
576  ProcessRunning = false;
577  return;
578  }
579  throw(extras::stacktrace_error("LoadDataLoop(): Recieved MxObject with _mxptr==nullptr and not at EOF"));
580  }
581 #ifdef _DEBUG
582  _ResPtrs.push_back(res.getmxarray());
583 #endif
584  std::lock_guard<std::mutex> rlock(_ResultsGroupMutex); //lock results
585  _ResultsGroup.move_back(res.releaseArray()); //move data to back of results list
586 
587 #ifdef _DEBUG
588  _ResultsPtrs.push_back(_ResultsGroup.getBack());
589 #endif
590  }
591  }
592  catch (...) {
593  ProcessRunning = false;
594  if (_Reader.isEOF()) { //just at end of file, no more data to read
595  _atEOF = true;
596  ProcessRunning = false;
597  return;
598  }
599  ErrorFlag = true;
600  std::lock_guard<std::mutex> lock(LastErrorMutex);
601  LastError = std::current_exception();
602  return;
603  }
604  ProcessRunning = false;
605  }
606 
608  virtual void StopProcessor() {
609  ProcessRunning = false;
610  std::this_thread::sleep_for(std::chrono::microseconds(1000)); //let some time pass
611  if (mThread.joinable()) {
612  mexPrintf("\tHalting AsyncMxFileReader");
613  mexEvalString("pause(0.2)");
614  mThread.join();
615  mexPrintf("...done.\n");
616  mexEvalString("pause(0.1)");
617  }
618  }
619 
621  virtual void StartProcessor() {
622  StopProcessor();
623  if (!_Reader.isFileOpen()) {
624  throw(extras::stacktrace_error("File is not open, cannot StartProcessor()"));
625  }
626  ErrorFlag = false;
627  ProcessRunning = true;
628  try {
629  mThread = std::thread(&AsyncMxFileReader::LoadDataLoop, this);
630  }
631  catch (...) {
632  ProcessRunning = false;
633  throw;
634  }
635  }

638  void clearLoadedData() {
639  std::lock_guard<std::mutex> rlock(_ResultsGroupMutex); //lock results
640  _ResultsGroup.clear();
641  }

648  }

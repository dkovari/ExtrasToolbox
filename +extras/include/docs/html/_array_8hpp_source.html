328  Array& concatenate(const ArrayBase<M>& b, size_t dim) {
329 
331  // Compute new size
332 
333  auto thisSz = this->size();
334  auto thatSz = b.size();
335  size_t thisSz_len = thisSz.size();
336  size_t thatSz_len = thatSz.size();
337 
338  size_t maxDimLen = std::max(std::max(thisSz.size(), thatSz.size()), dim + 1);
339 
340  // Loop over array dimensions and determine if sizes are compatible
341  thisSz.resize(maxDimLen);
342  thatSz.resize(maxDimLen);
343 
344  for (size_t j = 0; j < maxDimLen; j++) {
345 
346  if (j >= thisSz_len) {
347  thisSz[j] = 1;
348  }
349 
350  if (j >= thatSz_len) {
351  thatSz[j] = 1;
352  }
353 
354  if (j == dim) {
355  continue;
356  }
357 
358  if (thisSz[j] != thatSz[j]) {
359  throw(std::runtime_error("incompatible sizes for concatenate"));
360  }
361  }
362 
363  auto newSz = thisSz;
364  newSz[dim] = thisSz[dim] + thatSz[dim];
365 
367  // Create Temporary array to hold new data
368  Array<T> newArray(newSz);
369 
371  // Loop over dimension dim and copy data for all other dimensions
372 
373  size_t new_dim = 0; //accumulator for current position in newPtr
374  for (size_t d = 0; d < thisSz[dim]; d++) { //first loop over mxptr elements
375  //loop over other dims
376  for (size_t odim = 0; odim < newSz.size(); odim++) {
377  if (odim == dim) { //nothing to do for dim
378  continue;
379  }
380  if (thisSz[odim] == 1) {//nothing to do for singleton dimension
381  continue;
382  }
383  //loop over indecies of odim and copy
384  for (size_t od = 0; od < thisSz[odim]; od++) {
385  std::vector<size_t> subs(newSz.size(), 0.0); //create subscript array with all zeros
386  subs[odim] = od; //index for dimension being copied
387  subs[dim] = d; //index for dimension being concatenated
388 
389  newArray(subs) = (*this)(subs); //copy values
390 
391  }
392  }
393  new_dim++;
394  }
395 
396  for (size_t d = 0; d < thatSz[dim]; d++) { //second loop over end_obj elements
397  //loop over other dims
398  for (size_t odim = 0; odim < newSz.size(); odim++) {
399  if (odim == dim) { //nothing to do for dim
400  continue;
401  }
402  if (thatSz[odim] == 1) {//nothing to do for singleton dimension
403  continue;
404  }
405  //loop over indecies of odim and copy
406  for (size_t od = 0; od < thatSz[odim]; od++) {
407  std::vector<size_t> subs(newSz.size(), 0.0); //create subscript array with all zeros
408  subs[odim] = od; //index for dimension being copied
409  subs[dim] = d; //index for dimension being concatenated
410 
411  auto val = b(subs); //make tmp copy of data
412 
413  subs[dim] = new_dim; //change the sub, since newArray has different dim
414  newArray(subs) = val; //set value
415  }
416  }
417  new_dim++;
418  }
419 
420  moveFrom(newArray);
421 
422  return *this;
423  }

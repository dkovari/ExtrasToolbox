36  bool _isPersistent = false; //flag if data is persistent

39  //internal use, reshape real numeric or char array
40  template<bool copy_data> void reshape_real_char(const std::vector<size_t>& dims) {
41  if (_setFromConst) { //cannot reshape const
42  throw(extras::stacktrace_error("MxObject::reshape(): Cannot reshape MxObject linked to const mxArray*"));
43  }
44 
45  std::lock_guard<std::mutex> lock(_mxptrMutex); //lock _mxptr;
46 
48  if (_mxptr == nullptr) {
49  _mxptr = mxCreateNumericArray(dims.size(), dims.data(), mxDOUBLE_CLASS, mxREAL);
50  _managemxptr = true;
51  _setFromConst = false;
52  if (_isPersistent) {
53  mexMakeArrayPersistent(_mxptr);
54  }
55  return;
56  }
57 
58  //validate mxarray type
59  if (!(isnumeric() || ischar())) {
60  throw(extras::stacktrace_error("MxObject::reshape_real_char(): mxptr is not real numeric or char"));
61  }
62  if (iscomplex()) {
63  throw(extras::stacktrace_error("MxObject::reshape_real_char(): mxptr is not real numeric or char"));
64  }
65 
66  size_t oldNumel = numel();
67  size_t newNumel = extras::prod(dims);
68 
69  if (oldNumel >= newNumel) { //same size or smaller
70  size_t elsz = mxGetElementSize(_mxptr);
71  void* newData = mxRealloc(mxGetData(_mxptr), newNumel*elsz);
72  mxSetData(_mxptr, newData);
73  mxSetDimensions(_mxptr, dims.data(), dims.size()); //change dim size
74  if (_isPersistent) {
75  mexMakeMemoryPersistent(newData);
76  }
77  }
78  else { //bigger
79  mxArray* newPtr;
80  if (isnumeric()) {
81  newPtr = mxCreateNumericArray(dims.size(), dims.data(), mxGetClassID(_mxptr), mxREAL);
82  }
83  else {
84  newPtr = mxCreateCharArray(dims.size(), dims.data());
85  }
86  if (copy_data) {
87  std::memcpy(mxGetData(newPtr), mxGetData(_mxptr), mxGetElementSize(_mxptr)*std::min(mxGetNumberOfElements(_mxptr), mxGetNumberOfElements(newPtr)));
88  }
89  mxFree(mxGetData(_mxptr));
90  mxSetData(_mxptr, mxGetData(newPtr)); //set to new data pointer
91  mxSetData(newPtr, nullptr); //release pointer from newPtr's control
92  mxDestroyArray(newPtr);//delete newPtr
93  mxSetDimensions(_mxptr, dims.data(), dims.size()); //change dim size
94 
95  if (_isPersistent) {
96  mexMakeArrayPersistent(_mxptr);
97  }
98  }
99  }

148  newPtr = mxCreateNumericArray(dims.size(), dims.data(), mxGetClassID(_mxptr), mxREAL);

164  mxSetData(newPtr, nullptr); //release pointer from newPtr's control

172  mxSetDimensions(_mxptr, dims.data(), dims.size()); //change dim size

191  _managemxptr = true;
192  _setFromConst = false;
193  if (_isPersistent) {
194  mexMakeArrayPersistent(_mxptr);
195  }

197  }

199  //validate
200  if (!iscell()) {
201  throw(extras::stacktrace_error("MxObject::reshape_cell(): mxptr is not cell."));
202  }
203 
204  size_t oldNumel = numel();
205  size_t newNumel = extras::prod(dims);
206 
207  size_t elsz = mxGetElementSize(_mxptr);
208 
209  if (oldNumel >= newNumel) { //same size or smaller
210  for (size_t n = newNumel; n < oldNumel; n++) {
211  mxDestroyArray(mxGetCell(_mxptr,n));
212  }
213  void* data = mxRealloc(mxGetData(_mxptr), newNumel*elsz);
214  mxSetData(_mxptr, data);
215  if (_isPersistent) {
216  mexMakeMemoryPersistent(data);
217  }
218  }
219  else {
220  void* data = mxCalloc(newNumel, elsz);
221  if (copy_data) {
222  memcpy(data, mxGetData(_mxptr), elsz*oldNumel);
223  }
224  mxFree(mxGetData(_mxptr));
225  mxSetData(_mxptr, data);
226 
227  if (_isPersistent) {
228  mexMakeMemoryPersistent(data);
229  }
230  }
231  mxSetDimensions(_mxptr, dims.data(), dims.size());

235  template<bool copy_data> void reshape_struct(const std::vector<size_t>& dims) {
236  if (_setFromConst) { //cannot reshape const
237  throw(std::runtime_error("MxObject::reshape(): Cannot reshape MxObject linked to const mxArray*"));
238  }
239 
240  std::lock_guard<std::mutex> lock(_mxptrMutex); //lock _mxptr;
241 
243  if (_mxptr == nullptr) {
244  _mxptr = mxCreateStructArray(dims.size(), dims.data(), 0, nullptr);
245  _managemxptr = true;
246  _setFromConst = false;
247  if (_isPersistent) {
248  mexMakeArrayPersistent(_mxptr);
249  }
250  return;
251  }
252 
253  if (!isstruct()) {
254  throw(extras::stacktrace_error("MxObject::reshape_struct(): mxptr is not struct"));
255  }
256 
257  size_t oldNumel = numel();
258  size_t newNumel = extras::prod(dims);
259  size_t nfields = mxGetNumberOfFields(_mxptr);
260  size_t elsz = mxGetElementSize(_mxptr);
261  if (oldNumel >= newNumel) { //same size or smaller
262  for (size_t f = 0; f < nfields; f++) {
263  for (size_t n = newNumel; n < oldNumel; n++) {
264  mxDestroyArray(mxGetFieldByNumber(_mxptr, n, f));
265  }
266  void* data = mxRealloc(mxGetData(_mxptr), newNumel*nfields*elsz);
267  mxSetData(_mxptr, data);
268  if (_isPersistent) {
269  mexMakeMemoryPersistent(data);
270  }
271  }
272  }
273  else {
274  void* data = mxCalloc(nfields*newNumel, elsz);
275  if (copy_data) {
276  memcpy(data, mxGetData(_mxptr), elsz*oldNumel*nfields);
277  }
278  mxFree(mxGetData(_mxptr));
279  mxSetData(_mxptr, data);
280 
281  if (_isPersistent) {
282  mexMakeMemoryPersistent(data);
283  }
284  }
285  mxSetDimensions(_mxptr, dims.data(), dims.size());
286  };

314  virtual void copyFrom(const MxObject& src) {
315  std::lock_guard<std::mutex> lock(_mxptrMutex); //lock _mxptr;
316  deletemxptr_nolock();
317  if (src._mxptr == nullptr) {
318  _mxptr = nullptr;
319  _isPersistent = false;
320  _setFromConst = false;
321  }
322  else {
323  _mxptr = mxDuplicateArray(src._mxptr);
324  _managemxptr = true;
325  _setFromConst = false;
326  if (src._isPersistent) {
327  makePersistent();
328  }

707  void reshape(const std::vector<size_t>& dims) {
708  if (_setFromConst) { //cannot reshape const
709  throw(extras::stacktrace_error("MxObject::reshape(): Cannot reshape MxObject linked to const mxArray*"));
710  }
711 
712  // nullptr -> return numeric real double
713  if (_mxptr == nullptr) {
714  std::lock_guard<std::mutex> lock(_mxptrMutex); //lock _mxptr;
715  _mxptr = mxCreateNumericArray(dims.size(), dims.data(), mxDOUBLE_CLASS, mxREAL);
716  _managemxptr = true;
717  _setFromConst = false;
718  if (_isPersistent) {
719  mexMakeArrayPersistent(_mxptr);
720  }
721  return;
722  }
723 
724  // Perform resize
725  switch (mxGetClassID(_mxptr)) {
726  case mxDOUBLE_CLASS:
727  case mxSINGLE_CLASS:
728  case mxINT8_CLASS:
729  case mxUINT8_CLASS:
730  case mxINT16_CLASS:
731  case mxUINT16_CLASS:
732  case mxINT32_CLASS:
733  case mxUINT32_CLASS:
734  case mxINT64_CLASS:
735  case mxUINT64_CLASS:
736  case mxCHAR_CLASS:
737  if (iscomplex()) {
738  reshape_imag<true>(dims);
739  }
740  else {
741  reshape_real_char<true>(dims);
742  }
743  break;
744  case mxCELL_CLASS:
745  reshape_cell<true>(dims);
746  break;
747  case mxSTRUCT_CLASS:
748  reshape_struct<true>(dims);
749  break;
750  default:
751  throw(extras::stacktrace_error("reshape not implemented for class"));
752  }
753 
754  }

820  virtual MxObject& concatenate(const MxObject& end_obj, size_t dim) {

824  }

828  std::string("Cannot concatenate ") + std::string(mxGetClassName(_mxptr))
829  + std::string("with ") + std::string(mxGetClassName(end_obj))));
830  }
831  if (iscomplex() != end_obj.iscomplex()) {
832  throw(extras::stacktrace_error("concatenate(): either both arrays must be real or both must be complex"));
833  }
834 
836  // Struct NOT IMPLEMENTED ----> REMOVE WHEN FIXED

839  }

841  mxArray* end_ptr = (mxArray*)end_obj.getmxarray(); //mxArray* for end_obj
842  bool destroy_end_ptr = false;
843  if (&end_obj == this) { //cat with self, need to make a copy of the data
844  end_ptr = mxDuplicateArray(end_ptr);
845  destroy_end_ptr = true;

853  size_t thisSz_len = thisSz.size();
854  size_t thatSz_len = thatSz.size();
855

857 
858  // Loop over array dimensions and determine if sizes are compatible
859  thisSz.resize(maxDimLen);
860  thatSz.resize(maxDimLen);
861 
862  for (size_t j = 0; j < maxDimLen; j++) {
863 
864  if (j >= thisSz_len) {
865  thisSz[j] = 1;
866  }
867 
868  if (j >= thatSz_len) {
869  thatSz[j] = 1;
870  }
871 
872  if (j == dim) {
873  continue;
874  }
875 
876  if (thisSz[j] != thatSz[j]) {
877  throw(extras::stacktrace_error("incompatible sizes for concatenate"));
878  }

886  mxArray* newPtr = nullptr;
887  switch (mxGetClassID(_mxptr)) {
888  case mxDOUBLE_CLASS:
889  case mxSINGLE_CLASS:
890  case mxINT8_CLASS:
891  case mxUINT8_CLASS:
892  case mxINT16_CLASS:
893  case mxUINT16_CLASS:
894  case mxINT32_CLASS:
895  case mxUINT32_CLASS:
896  case mxINT64_CLASS:
897  case mxUINT64_CLASS:
898  case mxCHAR_CLASS:
899  {
900  if (mxIsComplex(_mxptr)) {
901  newPtr = mxCreateNumericArray(newSz.size(), newSz.data(), mxGetClassID(_mxptr), mxCOMPLEX);
902  }
903  else {
904  newPtr = mxCreateNumericArray(newSz.size(), newSz.data(), mxGetClassID(_mxptr), mxREAL);
905  }
906  }
907  break;
908  case mxCELL_CLASS:
909  newPtr = mxCreateCellArray(newSz.size(), newSz.data());
910  break;
911  case mxSTRUCT_CLASS:
912  throw(extras::stacktrace_error("concentate not implemented for struct"));
913  default:
914  throw(extras::stacktrace_error(std::string("concatenate not implemented for ")+std::string(mxGetClassName(_mxptr))));
915  }
916 
918  // Loop over dimension dim and copy data for all other dimensions
919 
920  size_t new_dim = 0; //accumulator for current position in newPtr
921  for (size_t d = 0; d < thisSz[dim]; d++) { //first loop over mxptr elements
922  //loop over other dims
923  for (size_t odim = 0; odim < newSz.size(); odim++) {
924  if (odim == dim) { //nothing to do for dim
925  continue;
926  }
927  if (thisSz[odim] == 1) {//nothing to do for singleton dimension
928  continue;
929  }
930  //loop over indecies of odim and copy
931  for (size_t od = 0; od < thisSz[odim]; od++) {
932  std::vector<size_t> subs(newSz.size(),0); //create subscript array with all zeros
933  subs[odim] = od; //index for dimension being copied
934  subs[dim] = d; //index for dimension being concatenated
936

938  subs[dim] = new_dim; //index for cat dim of new array is different
939  size_t dstIndex = mxCalcSingleSubscript(newPtr, subs.size(), subs.data());
940

946  memcpy(&dst[dstIndex*elm_sz], &src[srcIndex*elm_sz], elm_sz);
947  }
948  else { //complex

950  const uint8_t* src = (const uint8_t*)mxGetData(_mxptr);
951  uint8_t* dst = (uint8_t*)mxGetData(newPtr);
952  size_t elm_sz = mxGetElementSize(newPtr);
953  memcpy(&dst[dstIndex*elm_sz], &src[srcIndex*elm_sz], elm_sz);
954 #else

956  uint8_t* dst = (uint8_t*)mxGetData(newPtr);
957  size_t elm_sz = mxGetElementSize(newPtr);
958  memcpy(&dst[dstIndex*elm_sz], &src[srcIndex*elm_sz], elm_sz);
959 
960  src = (const uint8_t*)mxGetImagData(_mxptr);
961  dst = (uint8_t*)mxGetImagData(newPtr);
962  memcpy(&dst[dstIndex*elm_sz], &src[srcIndex*elm_sz], elm_sz);
963 #endif
964  }
965  }

968  mxSetCell(newPtr, dstIndex, mxGetCell(_mxptr, srcIndex));
969  mxSetCell(_mxptr, srcIndex, nullptr);
970  }
971  }
972  }
973  new_dim++;
974  }
975 
976  for (size_t d = 0; d < thatSz[dim]; d++) { //second loop over end_obj elements
977  //loop over other dims
978  for (size_t odim = 0; odim < newSz.size(); odim++) {
979  if (odim == dim) { //nothing to do for odim==dim, the outloop handles the copy on that dimension
980  continue;
981  }

983  continue;
984  }

995

1024  }
1025  }
1026  }

1028  }

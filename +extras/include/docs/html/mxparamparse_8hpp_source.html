91  int Parse(size_t nargs, const mxArray* pargs[]){
92 
93  if (nargs < 1) { return 0; }
94 
95  if(nargs%2 !=0){ //check that even number of inputs
96  mexPrintf("Incorrect number of arguments passed to MxInputParser::Parse\n");
97  return -1;
98  }
99 
100  for(size_t n=0;n<nargs-1;n+=2){
101  if(!mxIsChar(pargs[n])){
102  //not a string
103  mexPrintf("\n\targ[%d] is not a string\n",n);
104  mexPrintf("it is: %s\n",mxGetClassName(pargs[n]));
105  return 1;
106  }
107  //create key string
108  std::string key = getstring(pargs[n]);
109 
110  //Convert to upper case if not case sensitive
111  if(!_CaseSensitive) key = toupper(key);
112 
113  try{
114  Parameter.at(key) = pargs[n+1];
115  ParameterFound.at(key) = true;
116  }catch(std::out_of_range e){
117  //didn't find match
118  if(!IgnoreUnspecifiedParameters){
119  mexPrintf("arg[%d] could not find a match\n",n);
120  mexPrintf("\tusing key: '%s'\n",key.c_str());
121  mexPrintf("Possible Values:\n");
122  for(auto i:Parameter){
123  mexPrintf("\t'%s'\n",i.first.c_str());
124  }
125  return 2;
126  }
127  }
128  }
129  return 0;
130  }
